        /// <summary>
        /// 创建投射物，一般攻击用的走这
        /// 支持设置跟随目标，如果不设置则会走固定路径到TargetPos
        /// </summary>
        /// <param name="name"></param>
        /// <param name="birthPos"></param>
        /// <param name="scale"></param>
        /// <param name="target"></param>
        /// <param name="targetPos"></param>
        /// <param name="flySpeed"></param>
        /// <param name="angel"></param>
        /// <param name="animationCurve"></param>
        /// <param name="arriveAction"></param>
        /// <param name="endFunc"></param>
        /// <param name="autoCreateModel"></param>
        /// <returns></returns>
        public ProjectileHelper CreateProjectile(string name, Vector3 birthPos, float scale,Transform target,Vector3 targetPos, float flySpeed, float angel, AnimationCurve animationCurve, Action arriveAction,Func<bool> endFunc,bool autoCreateModel=true)

给定投射物名称（放在AB包中指定路径下），设置生成点位，缩放，追随目标，飞行速度，角度，动画曲线，到达事件，以及中断判定事件。如果追随目标不为空，则会在飞行过程中自动校正位置。
